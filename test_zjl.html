/api/clientActivity

REST API to retrieve client activities for observables

POST request example:

{
    "Triggers" : {
        "TimeRange" : 
        {
            "start" : 1466784000,
            "end" : 1467212400
        }
    },
    "Observables" : {
        "Client" :
        [
            {   
                "value":"1.1.1.1",
                "version":"ipv4"
            }
        ],
        "IpAddress" : 
        [
            {
                "value":"209.99.40.222", 
                "version": "ipv4"
            }
        ],
        "DomainName" : 
        [
            {
                "value":"www.theforwardview.com"
            },
            {
                "value":"dlnembnfbcpjnepmfjmngjenhhajpdfd.s3.amazonaws.com"
            }
        ],
        "Url" : 
        [
            {
                "value":"http://www.theforwardview.com/feed/"
            },
            {
                "value":"http://dlnembnfbcpjnepmfjmngjenhhajpdfd.s3.amazonaws.com/14/version.txt"
            }
        ]
    }
    "Configurations":
    {
        "logFormat":"src|srcport|dst|dstport|cs-uri"
    }
}

POST request example:

curl -X POST -H "Content-Type: application/json" --data '{ "Triggers" : {"TimeRange":{"start":1466784000,"end":1467212400}},"Observables":{"IpAddress":[{"value":"209.99.40.222", "version":"ipv4"}],"DomainName" :[{"value":"www.theforwardview.com"},{"value":"dlnembnfbcpjnepmfjmngjenhhajpdfd.s3.amazonaws.com"}],"Url" : [{"value":"http://www.theforwardview.com/feed/"} ,{"value":"http://dlnembnfbcpjnepmfjmngjenhhajpdfd.s3.amazonaws.com/14/version.txt"}]} "Configurations":{"logFormat":"src|srcport|dst|dstport|cs-uri"}}' localhost:8888/api/clientActivity
